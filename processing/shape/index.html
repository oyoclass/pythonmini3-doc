Example

from processing import *

def setup():
    size(400, 400)
    background(0, 0, 0)

def draw():
    fill(0, 0, 255)
    ellipse(56, 46, 55, 55)
    fill(255, 0, 0)
    triangle(130, 175, 158, 120, 186, 175)

run()
